04 — Серверная часть

Бэкенд: Node.js + Express

Серверная часть обрабатывает запросы к API, управляет маршрутизацией и проверяет уровень доступа для каждого запроса.

  • Асинхронная модель Event Loop
  • REST API с CRUD-операциями
  • Middleware-защита маршрутов
  • Загрузка файлов через Multer
  • Централизованная обработка ошибок
assignmentController.js 
const getAssignments = async (req, res, next) => {
  const filter = {};
  if (req.query.search) {
    filter.title = { $regex: req.query.search };
  }
  // Фильтрация по классу ученика
  if (req.user.role === 'student') {
    filter.targetGrade = req.user.grade;
  }
  const assignments = await Assignment
    .find(filter)
    .populate('subject', 'name icon')
    .sort({ createdAt: -1 });
  res.json({ success: true, data: assignments });
};
05 — База данных

MongoDB + Mongoose

models/User.js 
const userSchema = new mongoose.Schema({
  firstName: {
    type: String,
    required: [true, 'Имя обязательно'],
    maxlength: [50, 'Макс 50 символов'],
  },
  email: {
    type: String, unique: true,
    match: [/^\S+@\S+\.\S+$/, 'Email некорректен'],
  },
  role: {
    type: String,
    enum: ['student', 'teacher', 'admin'],
  },
  grade: {
    type: Number,
    min: [1, 'Класс от 1'],
    max: [11, 'Класс до 11'],
  },
});

// Автоматическое хэширование пароля
userSchema.pre('save', async function() {
  this.password = await bcrypt.hash(this.password, 10);
});

Гибкая NoSQL-база данных MongoDB хранит данные в JSON-документах. Mongoose накладывает строгую валидацию.

📄 5 Mongoose-моделей
User, Assignment, Submission, Subject, Schedule — полная структура данных школьной платформы
🔗 Связи (populate)
Assignment → Subject + Teacher, Submission → Student + Assignment
🛡️ Валидация
Класс строго от 1 до 11, email уникален, роли ограничены enum
06 — Безопасность

JWT + Bcrypt: Защита данных

Каждый API-запрос проходит двойную проверку. Пароли хранятся в виде необратимого хэша.

authMiddleware.js
Проверяет JWT-токен из заголовка Authorization. Подделанный или просроченный токен → 401 Unauthorized
roleMiddleware.js
Ученик не может удалить задание → 403 Forbidden
errorHandler.js
Единый обработчик ошибок: ValidationError, CastError, дубли, JWT-сбои
errorHandler.js 
const errorHandler = (err, req, res, next) => {
  let statusCode = err.statusCode || 500;
  let message = err.message;

  if (err.name === 'ValidationError') {
    statusCode = 400;
    message = Object.values(err.errors)
      .map(e => e.message).join(', ');
  }

  if (err.code === 11000) {
    statusCode = 400;
    message = `Поле уже существует`;
  }

  if (err.name === 'JsonWebTokenError') {
    statusCode = 401;
    message = 'Невалидный токен';
  }

  res.status(statusCode).json({
    success: false, error: message
  });
};
